    客户端使用说明

    1.census下载

    从golib上下载包census，如使用http服务，则在应用中配置如下：

    [census]
Host = "127.0.0.1"
Port = 8013

    2.封装函数：

    1.添加任务:

    type Cli interface {
    //任务请求成功回调
    OnSuccess(func())
    //任务请求失败回调
    OnError(func())
    //请求传参
    //appName:应用名称,目前支持：aiclass
    //action:每个action对应一个任务,参照项目协议
    //data:该action传参所需要参数
    SetRequest(appName string, action int, data interface{}, callType int) error
    //绑定参数：如：http绑定回调类型，mq绑定某一队列
    Bind(key int, value interface{})
    //执行，内部使用go-routine,非阻塞
    Execute()
    //获取响应详情
    GetResponse() *response
    //获取错误详情
    GetError() error
}

    目前已支持http，rabbitMq, nsq 进行添加任务，代码示例如下：（注意：census,rabbitMq,nsq必须进行相应的配置）

        //传参
    data := map[string]interface{}{
        "classId":  request.ClassId,
        "courseId": request.CourseId,
    }
    //1.http请求
    //http protocal
        //绑定回调类型，枚举: 0,返回taskUUId;1,返回任务执行的结果
        client := census.NewHttpClient(census.CallTypeDirect)
    client.SetRequest("aiclass", 101, data)
    //成功回调
    client.OnSuccess(func() {
        //获取响应结构体
        resp := client.GetResponse()
        if resp != nil {
            fmt.Printf("get success response taskUUId:[%v],action:[%v],appName:[%v],detail:[%v]", resp.TaskUUId, resp.Action, resp.AppName, resp.TaskInfo)
        }
    })
    //失败回调
    client.OnError(func() {
        err := client.GetError()
        if err != nil {
            fmt.Printf("census error: %v", err.Error())
        }
    })
    //执行
    client.Execute()
    
    //2.rabbit-mq请求
    //mq protocal 绑定队列
    client := census.NewRabbitMqClient("aiclass_student")
    client.SetRequest("aiclass", 101, data)
    //成功回调
    client.OnSuccess(func() {
        //消息队列没有结构体响应
        //此处可以封装业务
        ...
    })
    //失败回调
    client.OnError(func() {
        err := client.GetError()
        if err != nil {
            fmt.Printf("census error: %v", err.Error())
        }
    })
    //执行
    client.Execute()
        //3.nsq请求
    //nsq protocal 绑定topic
    client := census.NewNsqClient("nsqTopicAiclass")
    client.SetRequest("aiclass", 101, data)
    //成功回调
    client.OnSuccess(func() {
        //消息队列没有结构体响应
        //此处可以封装业务
        ...
    })
    //失败回调
    client.OnError(func() {
        err := client.GetError()
        if err != nil {
            fmt.Printf("census error: %v", err.Error())
        }
    })
    //执行
    client.Execute()

    (2) 获取任务详情

    func GetTaskInfo(taskUUId string) (resp *Response, err error)
传参说明：
taskUUId:任务唯一id

    代码示例如下：

        //获取任务详情
    
    ...you have got taskUUId ...
    
    resp, err := census.GetTaskInfo(taskUUId)
    if err != nil {
        logger.Error("census", helper.GenerateUUID(), "http get task info fail: "+err.Error())
        return
    }
    logger.Info("census", helper.GenerateUUID(), "http get task info success: detail:", resp)
